##第二步：gulpfile.js文件的配置：
一般所有的任务都是在该文件中配置，然后在命令行中执行该文件中配置的任务。格式如下：

//一：导入gulp模块之后，才可以使用gulp命令
var gulp=require('gulp');


二:单个任务配置
gulp.task('任务名',function(){
    gulp.src('')//src中一般是要操作的文件的路径
    .pipe();//pipe一般是对文件的具体操作。
})
//然后在命令中执行如下命令：
gulp 任务名;


//三:多个任务，即任务队列的执行（即把所有的自定义任务全部添加到默认任务中）
gulp.task('default',['copy-index','copy-images','webserver'],function(){
    console.log('任务队列执行完毕');
});
//然后在命令行中执行如下命令：
gulp; 
或者
gulp default;

//四：任务处理程序
gulp.task('任务名',funciton(){
    gulp.src(源文件)
    .pipe(对源文件的处理操作)
    .pipe(把处理后的文件输出到指定位置)
})

##三：gulpfile.js常见任务的实现：

###文件拷贝任务
1：拷贝单个文件
gulp.task(‘copyindex’,function(){
gulp.src(‘./index.html’).pipe(gulp.dest(‘./build/‘));
});
2：拷贝文件夹
gulp.src(‘./images/.{png,jpg}’)
拷贝当前文件以及子目录等所有文件
// gulp.src(‘./images/**/‘)
.pipe(gulp.dest(‘./build/images/‘));
});
3：同时拷贝多个文件
gulp.task(‘copydata’,function(){
gulp.src([‘./index.html’,’./images//*’])
.pipe(gulp.dest(‘./build/data’));
});
4：拷贝多个文件（同时过滤部分文件）
gulp.task(‘copypartdata’,function(){
gulp.src([‘./index.html’,’./images//‘,’!./images/.jpg’])
.pipe(gulp.dest(‘./build/data’));
});

    说明：gulp.src指定要操作的源文件，可以是单个字符，也可以是数组
         gulp.dest()指定修改之后的文件的输出路径。
         拷贝多个文件就用[]来引用多个文件
         过滤部分文件就用感叹号！来实现。


5：如果以上四个任务同时执行，该如何实现呢？
gulp.task('copy',['copyindex','copyimages','copydata','copypartdata'],function(){
    console.log("任务队列执行完毕");
})
    注意点：此时四个任务是同时执行，即并行执行，当然我们也可以通过sequence设置成串行执行

6：任务监听：当任务比较多个时候，每次修改了任务，都需要我们重新执行一次，
才可以使目标文件也得以改变，这样就会很麻烦，这时，我们可以为多个任务设置监听，
这样只要任务对应的源文件修改，该监听任务就会被触发，这样目标文件也会自动修改。
gulp.task('watch',function(){
    gulp.watch('./index.html',['copyindex']);
    gulp.watch('./images/**/*',['copyimage']);
    gulp.watch(['./index.html',[./images/**/*]],['copydata']);
});

然后我们命令行中，只需开启watch任务，它就会在后台一直运行着，监听指定任务的变化

####第二步：在gulpfile.js中引入插件

//引入sass压缩模块
var minifyCSS=require(‘gulp-minify-css’);

####第三步：创建sass预编译以及压缩任务
var cssFiles=[
‘./src/styles/app.scss’//scss预编译的源文件，可以是多个
];
gulp.task(‘scss’,function(){
gulp.src(cssFiles)
.pipe(sass().on(‘error’,sass.logError))//注意必须要输出错误日志

##五:gulp本地服务插件gulp-webserver

####第一步：安装插件
cnpm install gulp-webserver;

####第二步：引入插件
var webserver=require(‘gulp-webserver’);

####第三步：创建插件，同时进行相关配置
gulp.task(‘webserver’,function(){
gulp.src(‘./‘)//指定在哪个文件夹下开启服务
.pipe(webserver({
host:’localhost’,
port:80,
directoryListing:{
enable:true,//表示浏览器访问的时候，可以显示目录
path:’./‘ //表示浏览器访问的时候，显示指定目录
},
livereload:true//表示当内容更改时，浏览器自动刷新
}));
});

####第四步：开启任务
gulp webserver;//开启服务（按ctrl+c退出本地服务）

##六：文件合并与压缩（js文件）

####第一步：安装插件
cnpm install gulp-concat -D//合并
cnpm install gulp-uglify -D//压缩

####第二步：引入插件
/引入contact模块
var concat=require(‘gulp-concat’);

//引入js压缩模块
var uglify=require('gulp-uglify');

####第三步：创建任务
gulp.task(‘concat’,function(){
gulp.src([‘./myscripts/first.js’,’./myscripts/second.js’])//要合并的文件放在数组中
.pipe(concat(‘vender.js’))//合并文件
.pipe(uglify())//将合并后的文件再压缩
.pipe(gulp.dest(‘./build/js/‘));//输出文件
});

注意：此时压缩后的文件名也为wender.js，会把原来合并的文件覆盖掉，
    那么如何才能保证合并后的文件和压缩后的文件同时存在呢？就需要用到另个插件：gulp-rename;

    gulp.task('concat',function(){
        gulp.src(['./myscripts/first.js','./myscripts/second.js'])
        .pipe(concat('vender.js'))
        .pipe(gulp.dest('.build/js/'))//合并后先输出一次
        .pipe(rename('vender.min.js'))//重命名
        .pipe(uglify())//压缩
        .pipe(gulp.dest('./build/js/'));//压缩后在输出一次
    });

#####第四步：开启任务
gulp concat;//开启任务

#七:webpack使用

####1：安装插件
cnpm install webpack -g;//注意webpack全局和本地都要安装

####2：常见用法：

#####文件打包：

*    文件打包：webpack 源文件  目标文件

#####webpack模块化：

1. 导入js模块：即使用commonjs中的require()导入指定模块。
2. 导入css模块：使用css-loader和style-loader模块导入
      * 安装插件：$ cnpm install css-loader style-loader -D
      * 在js中导入插件：require('style!css!./index.css');

#####webpack.config.js配置文件的配置

######1：webpack.config.js文件的配置：
module.exports={
entry:’./entry.js’,//定义入口文件
output:{//定义输出文件的路径和文件名
path:__dirname,
filename:’bundle.js’
},
module:{
//定义加载器，是一个数组
loaders:[
{
//表示所有.css文件都是以style!css加载器去加载。
test:/.css/,
loader:’style!css’
}
]
}
}

######2：配置号webpack配置文件之后，我们在js中导入css模块的时候，就不需要再指定加载器
require(‘./index.css’);
注意：如果没有配置webpack.config.js文件，则导入css模块的时候，必须配置加载器：即
require(‘style!css!./index.css’);

######3.sourcemap的配置：
作用：主要用于我们在测试的时候，可以在浏览器的source标签下查看源代码

默认情况下，如果没有配置source-map，则浏览器下只会显示：出入文件index.html，
而那些name.js等等其他模块文件时不会显示的。

那么如何才能让这些文件也显示出来呢？
执行如下命令：webpack --devtool source-map
或者在配置文件中配置：dev-tool:'sourcemap'.这样我们在执行的时候，将entry.js文件
打包成bundle.js的同时，将其他文件也会显示到浏览器的source标签下。

##八：gulp与webpack的结合使用

#####第一步：安装插件
npm install viny-named -D;//文件名模块
npm install gulp-webpack -D;//js打包模块
npm install gulp-uglify -D;//js压缩模块
npm install imports-loader -D;//webpack中的加载器模块

#####第二步：引入插件
/引入js文件名模块
var named=require(‘vinyl-named’);
//引入webpack模块
var webpack=require(‘gulp-webpack’);
//引入js压缩模块
var uglify=require(‘gulp-uglify’);
//注意：imports-loader模块不需要引入，直接在webpack中使用

#####第三步：创建任务
//js源文件
var jsFiles=[
‘./src/scripts/app.js’//可以配置多个文件
]

gulp.task('packjs', function () {
  gulp.src(jsFiles)
    .pipe(named())//将源文件的文件名加到管道流中（在打包输出时会用到name）
    .pipe(webpack({
      output: {
        filename: '[name].js'//此时name就表示加入管道流中的文件名(这样输出文件和源文件的文件名保持一致。)
      },
      module: {
        loaders: [
          {
            test: /\.js$/,
            loader: 'imports?define=>false'

          }
        ]
      }
    }))
    .pipe(uglify().on('error', function (err) {//压缩文件
      console.log('\x07', err.lineNumber, err.message);
      return this.end();
    }))
    .pipe(gulp.dest('./build/prd/scripts/'));//输出文件
});

#####第四步：开启任务
gulp packjs;//开启任务

##九：mock数据获取

#####1：引入插件
var url=require(‘url’);//用于url的读取和设置
var fs=require(‘fs’);//用于对mock文件的读取等操作
注意：这两个文件是node.js中自带组件，不需要npm安装

#####2：加入到本地服务中
gulp.task(‘webserver’,function(){
gulp.src(‘./‘)
.pipe(webserver({
host:’localhost’,
port:80,
directoryListing:{
enable:true,
path:’./‘
},
livereload:true,

        //读取mock数据
        middleware:function(req,res,next){
            var urlObj=url.parse(req.url,true);//获取请求url，并且封装成url对象
            var method=req.method;//获取请求方式
            switch(urlObj.pathname){
                case '/api/orders/':
                    res.setHeader("Content-Type",'application/json');
                    fs.readFile('./mock/orders.json','utf-8',function(err,data){
                        res.end(data);//返回给浏览器的数据
                        console.log(data);//输出到控制台中
                    });
                    break;
                case '/api/goods/':
                    break;
                case '/api/images':
                    break;
            }
            next();
        }

    }));
});

#####3：开启本地服务
gulp;//开启本地服务

##10：gulp版本控制

#####第一步：安装插件
npm install gulp-rev -D;//生成版本号
npm install gulp-rev-collector -D;//映射版本号

#####第二步：引入插件

#####第三步：创建任务

#######创建任务一：生成版本号文件
gulp.task(‘rev’,function(){
gulp.src(cssDestFiles)
.pipe(rev())//添加版本号即一段md5串
.pipe(gulp.dest(‘./build/prd/styles/‘))
.pipe(rev.manifest())
.pipe(gulp.dest(‘./ver/styles’));

    gulp.src(jsDestFiles)
        .pipe(rev())//添加版本号即一段md5串
        .pipe(gulp.dest('./build/prd/scripts/'))
        .pipe(rev.manifest())
        .pipe(gulp.dest('./ver/scripts'));
});

########创建任务二：根据版本映射文件，将html中的js与css自动更新成带版本号的文件名。
gulp.task(‘revhtml’,function(){
gulp.src([‘./ver/*/.json’,’./*.html’])
.pipe(revCollector())//将两者映射起来，根据映射文件修改html文件js和css的版本号
.pipe(gulp.dest(‘./‘));
});

#####第四步：开启任务
gulp rev;
gulp revhtml;
